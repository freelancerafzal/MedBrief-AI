// Ensure this structure is used for every turn in the conversation async function handleSendMessage(query, chatSession, currentDocumentContext) { // Construct the payload as an array of parts const messageParts = [{ text: query }]; // If a document was uploaded, re-inject it into this turn's payload if (currentDocumentContext) { messageParts.push(currentDocumentContext); } try { // Correct SDK usage: send the array of parts directly const response = await chatSession.sendMessage(messageParts); return response.text; } catch (error) { console.error("Pipeline Error:", error); throw error; } }
